SOV_higher_armor_tank_production = {
	enable = {
		original_tag = SOV
		has_tech = basic_medium_tank
	}
	abort = {
		OR = {
			has_tech = main_battle_tank
			AND = {
				has_war = yes
				surrender_progress > 0.7
			}
		}
	}

	ai_strategy = {	type = role_ratio		id = motorized		value = 15	}
	ai_strategy = {	type = role_ratio		id = armor				value = 20	}
	ai_strategy = {	type = role_ratio		id = infantry			value = 63	}

}

SOV_increase_lspart_production = {
	enable = {
		original_tag = SOV
		num_of_military_factories > 50
		NOT = { has_equipment = { light_tank_artillery_equipment > 500 } }
	}
	abort = {
		NOT = {
			AND = {
				num_of_military_factories > 75
				NOT = { has_equipment = { light_tank_artillery_equipment > 500 } }
			}
		}
	}

	ai_strategy = { type = equipment_production_min_factories  id = artillery value = 10 }
	ai_strategy = { type = equipment_variant_production_factor id = light_tank_artillery_equipment value = 20 }
}
