	id = soviet.512
	picture = GFX_report_event_soviet_soldiers
	title = soviet.512.t
	desc = soviet.512.d

  trigger = {
    tag = PER
    has_war_with = ENG
    surrender_progress > 0.25
    has_civil_war = no
    OR = {
      has_global_flag = countenance_twin
      has_global_flag = countenance_alone
    }
  }

  mean_time_to_happen = {	days = 1 }

	fire_only_once = yes

  option = {
    name = soviet.512.a

    PER = {
      if = {
        limit = {
          is_in_faction = yes
        }
        leave_faction = yes
      }
    }

    ENG = {
      white_peace = {
        tag = PER
        message = PER_agree_peace1
      }
      country_event = {
        id = soviet.513
        days = 1
      }

    }
    if = {
      limit = {
        has_global_flag = countenance_twin
        SOV = { has_war_with = PER }
      }
      SOV = {
        white_peace = {
          tag = PER
          message = PER_agree_peace2
        }
        country_event = {
          id = soviet.514
          days = 1
        }
      }
    }
    set_global_flag = op_countenance_completed
  }
}
